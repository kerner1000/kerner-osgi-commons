package de.jcommandlineparser;

import java.util.Arrays;
import java.util.List;

/**
 * <p>
 * Example use of <code>CmdLineParser</code>
 * </p>
 * 
 * @author Alexander Kerner
 * @lastVisit 2009-12-26
 * 
 */
public class Test {
  
  private final static CmdLineParser parser = new CmdLineParserImpl();

  public static void main(String[] args) {
    try {
      System.out.println("Arguments are: " + Arrays.asList(args));      

      // Use OptionBuilder class to build new Options
      // StringOption
      OptionBuilder.buildNewOption("string", 's');
      OptionBuilder
          .setDescription("some option with string value");
      final StringOption stringOption = OptionBuilder.getNewStringOption();
      
      // IntegerOption
      OptionBuilder.buildNewOption("intger", 'i');
      OptionBuilder.setRequired();
      final IntegerOption integerOption = OptionBuilder.getNewIntegerOption();
      
      // BooleanOption
      OptionBuilder.buildNewOption("boolean", 'b');
      final BooleanOption booleanOption = OptionBuilder.getNewBooleanOption();
      
      // VoidOption
      OptionBuilder.buildNewOption("void", 'v');
      final VoidOption voidOption = OptionBuilder.getNewVoidOption();
      
      
      
      // Register created options to parser
      parser.registerOption(stringOption);
      parser.registerOption(integerOption);
      parser.registerOption(booleanOption);
      parser.registerOption(voidOption);

      
      // Parser returns not recognized command line arguments after
      // parsing
      final String[] unknown = parser.parse(args);

      /*
       * "getValues()" will return a list of values for an option. use
       * "getValues().get(0)" if you expect the option to have only one
       * value.
       */
      final List<String> stringValues = parser.getValues('s', String.class);
      final boolean stringIsSet = parser.isSet('s');
      
      final List<Integer> integerValues = parser.getValues('i', Integer.class);
      boolean integerIsSet = parser.isSet('i');
      
      final List<Boolean> booleanValues = parser.getValues('b', Boolean.class);
      final boolean booleanIsSet = parser.isSet('b');
      
      final List<Void> voidValues = parser.getValues('v', Void.class);
      final boolean voidIsSet = parser.isSet('v');
      

      System.out.println("Option " + stringOption + ": Value: " + stringValues + ", Set: " + stringIsSet);
      System.out.println("Option " + integerOption + ": Value: " + integerValues + ", Set: " + integerIsSet);
      System.out.println("Option " + booleanOption + ": Value: " + booleanValues + ", Set: " + booleanIsSet);
      System.out.println("Option " + voidOption + ": Value: " + voidValues + ", Set: " + voidIsSet);

      System.out.println("unknown command line arguments: "
          + Arrays.toString(unknown));

      System.exit(0);

    } catch (Throwable t) {
      System.err.print(t.toString());
      System.exit(1);
    }
  }
}
